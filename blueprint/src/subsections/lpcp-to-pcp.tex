\begin{lemma}[Chernoff bound]
\label{LPCPToPCP.chernoff_bound}
\lean{LPCPToPCP.chernoff_bound}
\leanok
Let \(X_1,\dots, X_N\) be independent random variables over \(\{0,1\}\), and
let \(\mu=\EE[\sum_{i=1}^N X_i]\). For any \(\Delta > 0\),
\[
  \Pr\left[\sum_{i=1}^N X_i \ge \mu+\Delta\right]
  \le \exp(-2\Delta^2/N).
\]
\end{lemma}
\begin{proof}\leanok
For \(\lambda>0\), Hoeffding's elementary exponential-moment estimate gives
\[
  \EE[\exp(\lambda(X_i-\EE[X_i]))]\le \exp(\lambda^2/8).
\]
Independence therefore implies
\[
  \EE\left[\exp\left(\lambda\left(\sum_i X_i-\mu\right)\right)\right]
  \le \exp(N\lambda^2/8).
\]
Markov's inequality bounds the tail by
\(\exp(-\lambda\Delta+N\lambda^2/8)\); choosing
\(\lambda=4\Delta/N\) gives the result.
\end{proof}

\begin{definition}
\label{LPCPToPCP.numShifts}
\lean{LPCPToPCP.numShifts}
\leanok
For an LPCP verifier making at most \(q\) linear queries, set
\[
  t := 8\bigl(\lfloor \log_2(100q)\rfloor+1\bigr).
\]
% Lecture 08 only needs \(t=O(\log q)\). The Lean proof fixes this base-two
% value because the formalization uses `Nat.log2`; it is stronger than needed
% for \(q\exp(-t/8)\le 1/100\).
This is the concrete logarithmic shift count used in Lean. The converted PCP
verifier uses \(t\) independent random shifts to decode each linear query by
plurality.
\end{definition}

\begin{definition}
\label{LPCPToPCP.encodedProof}
\lean{LPCPToPCP.encodedProof}
\uses{PCPVerifier}
\leanok
For a linear LPCP proof \(\pi\in\F_2^\ell\), its PCP encoding is the truth
table
\[
  \widehat{\pi}_{\mathrm{lin}}(a) := \langle \pi,a\rangle,
  \qquad a\in\F_2^\ell.
\]
\end{definition}

\begin{definition}
\label{LPCPToPCP.correctedAnswer}
\lean{LPCPToPCP.correctedAnswer,LPCPToPCP.decodedLinearQuery,LPCPToPCP.pluralityZMod2}
\uses{LPCPToPCP.numShifts}
\leanok
Given a PCP proof \(\widehat{\pi}:\F_2^\ell\to\F_2\), shifts
\(r_1,\dots,r_t\in\F_2^\ell\), and a linear query \(a\in\F_2^\ell\), define
the locally corrected answer
\[
  \mathsf{Corr}_{\widehat{\pi}}(a;r_1,\dots,r_t)
  :=
  \operatorname{plurality}_{i\in[t]}
  \bigl(\widehat{\pi}(a+r_i)-\widehat{\pi}(r_i)\bigr).
\]
\end{definition}

\begin{definition}
\label{LPCPToPCP.verifier}
\lean{LPCPToPCP.verifier}
\uses{PCPVerifier,LPCPVerifier,BLR.basicVerifier,LPCPToPCP.correctedAnswer}
\leanok
Let \(V\) be an LPCP verifier over \(\F_2\). The converted PCP verifier
\(\overline V\), with oracle access to
\(\widehat{\pi}:\F_2^\ell\to\F_2\), proceeds as follows on input \(x\):
\begin{enumerate}
  \item Run the BLR verifier \ref{BLR.basicVerifier} on \(\widehat{\pi}\). If
    it rejects, reject.
  \item Sample \(t\) shifts \(r_1,\dots,r_t\in\F_2^\ell\), with \(t\) as in
    \ref{LPCPToPCP.numShifts}.
  \item Simulate \(V(x)\). Whenever \(V\) asks a linear query
    \(a\in\F_2^\ell\), answer it by
    \(\mathsf{Corr}_{\widehat{\pi}}(a;r_1,\dots,r_t)\).
\end{enumerate}
\end{definition}

\begin{lemma}
\label{LPCPToPCP.verifier_completeness}
\lean{LPCPToPCP.verifier_completeness}
\uses{LPCPToPCP.verifier,LPCPToPCP.encodedProof,BLR_basic_completeness_ZMod2,LPCP}
\leanok
If \(V\) has completeness error \(\varepsilon_c\) on \(x\) with linear proof
\(\pi\), then \(\overline V\) has the same completeness guarantee on \(x\) with
PCP proof \(\widehat{\pi}_{\mathrm{lin}}\).
\end{lemma}
\begin{proof}\leanok
The encoded proof is linear, so the BLR test accepts with probability \(1\) by
\ref{BLR_basic_completeness_ZMod2}. For every query \(a\) and every shift
\(r_i\),
\[
  \widehat{\pi}_{\mathrm{lin}}(a+r_i)-\widehat{\pi}_{\mathrm{lin}}(r_i)
  = \langle \pi,a\rangle.
\]
Thus every plurality answer equals the corresponding LPCP oracle answer, and
the simulation has exactly the same acceptance probability as \(V\).
\end{proof}

\begin{lemma}
\label{LPCPToPCP.local_correction}
\lean{LPCPToPCP.localCorrectionBad_single_le_quarter_of_not_far,LPCPToPCP.correctedAnswer_failure_le_exp_of_single,LPCPToPCP.chernoff_half_of_mean_le_quarter,LPCPToPCP.numShifts_eta_bound}
\uses{LPCPToPCP.correctedAnswer,LPCPToPCP.chernoff_bound}
\leanok
Suppose \(\widehat{\pi}\) is within relative distance \(<1/8\) of the linear
function \(u\mapsto\langle\pi,u\rangle\). For every query
\(a\in\F_2^\ell\),
\[
  \Pr_{r_1,\dots,r_t}\!\left[
    \mathsf{Corr}_{\widehat{\pi}}(a;r_1,\dots,r_t)\neq \langle\pi,a\rangle
  \right]
  \leq \exp(-t/8).
\]
\end{lemma}
\begin{proof}\leanok
For one shift \(r\), the corrected sample can be wrong only if
\(\widehat{\pi}(a+r)\) or \(\widehat{\pi}(r)\) disagrees with the nearby linear
function. A union bound gives error probability at most
\(2\cdot(1/8)=1/4\). If \(X_i\) is the indicator of failure for shift \(r_i\),
then \(\EE[\sum_i X_i]\le t/4\). The plurality answer is wrong only if
\(\sum_i X_i\ge t/2\), so \ref{LPCPToPCP.chernoff_bound} with
\(\Delta=t/4\) gives \(\exp(-t/8)\).
% Lecture 07 states a looser concentration estimate with different constants.
% The blueprint follows the Hoeffding normalization used by the Lean lemma
% `chernoff_half_of_mean_le_quarter`, which is mathematically stronger here.
\end{proof}

\begin{lemma}
\label{LPCPToPCP.corrected_simulation_hybrid}
\lean{LPCPToPCP.corrected_run_accept_le_linear_add,LPCPToPCP.verifier_accept_le_linear_add_of_query_correction}
\uses{LPCPToPCP.verifier,LPCPToPCP.local_correction,LPCP}
\leanok
Assume every possible linear query \(a\) is locally corrected incorrectly with
probability at most \(\eta\). If the simulated LPCP computation makes at most
\(q\) proof queries, then its acceptance probability under the corrected PCP
oracle is at most the acceptance probability under the nearby linear proof
plus \(q\eta\).
\end{lemma}
\begin{proof}\leanok
Expose the LPCP computation one oracle query at a time. At a proof query, the
two simulations can diverge only when the local correction for the queried
linear form is wrong; by assumption this has probability at most \(\eta\).
Inducting over the query-bound derivation gives an additive loss of at most
\(q\eta\).
% Lecture 08 presents this step as a union bound over the verifier's \(q\)
% queries. The Lean proof phrases the same estimate as a sequential hybrid,
% which also covers adaptive LPCP verifiers whose later queries may depend on
% earlier corrected answers.
\end{proof}

\begin{lemma}
\label{LPCPToPCP.verifier_soundness_far_case_of_isFar}
\lean{LPCPToPCP.verifier_soundness_far_case_of_isFar}
\uses{LPCPToPCP.verifier,BLR_basic_soundness_ZMod2}
\leanok
If \(\widehat{\pi}\) has relative distance at least \(1/8\) from every linear
function, then
\[
  \Pr[\overline V^{\widehat{\pi}}(x)=1]\leq 7/8 .
\]
\end{lemma}
\begin{proof}\leanok
By \ref{BLR_basic_soundness_ZMod2}, the initial BLR test rejects with
probability at least \(1/8\). Since \(\overline V\) accepts only when this test
accepts, its acceptance probability is at most \(7/8\).
\end{proof}

\begin{lemma}
\label{LPCPToPCP.verifier_soundness_near_case}
\lean{LPCPToPCP.verifier_soundness_near_case_of_single_query_bound}
\uses{LPCPToPCP.verifier,LPCPToPCP.local_correction,LPCPToPCP.corrected_simulation_hybrid,LPCP}
\leanok
If \(x\notin L\), \(V\) has LPCP soundness \(\varepsilon_s\), and
\(\widehat{\pi}\) is within distance \(<1/8\) of some linear function
\(u\mapsto\langle\pi,u\rangle\), then
\[
  \Pr[\overline V^{\widehat{\pi}}(x)=1]\leq \varepsilon_s+\frac{1}{100}.
\]
\end{lemma}
\begin{proof}\leanok
By \ref{LPCPToPCP.local_correction}, every possible linear query has local
correction error at most \(\eta=\exp(-t/8)\). The hybrid estimate
\ref{LPCPToPCP.corrected_simulation_hybrid} compares the corrected simulation
with the simulation using the nearby linear proof and loses at most
\(q\eta\). The choice of \(t\) in \ref{LPCPToPCP.numShifts} gives
\(q\eta\le 1/100\). The nearby linear proof is a valid LPCP proof, so its
acceptance probability is at most \(\varepsilon_s\) because \(x\notin L\).
\end{proof}

\begin{lemma}
\label{LPCPToPCP.verifier_soundness_of_cases}
\lean{LPCPToPCP.verifier_soundness_of_cases}
\uses{LPCPToPCP.verifier_soundness_far_case_of_isFar,LPCPToPCP.verifier_soundness_near_case}
\leanok
For every \(x\notin L\) and every PCP proof \(\widehat{\pi}\),
\[
  \Pr[\overline V^{\widehat{\pi}}(x)=1]
  \leq \max\{7/8,\varepsilon_s+1/100\}.
\]
\end{lemma}
\begin{proof}\leanok
If \(\widehat{\pi}\) is \(1/8\)-far from linear, apply
\ref{LPCPToPCP.verifier_soundness_far_case_of_isFar}. Otherwise choose a
linear function within distance \(<1/8\) and apply
\ref{LPCPToPCP.verifier_soundness_near_case}.
\end{proof}

\begin{lemma}
\uses{LPCP,PCP,LPCPToPCP.verifier}
\label{LPCP_to_PCP_ZMod2}
\lean{LPCP_to_PCP_ZMod2}
\leanok
\[
\LPCP[\varepsilon_c,\varepsilon_s,\Sigma=\F_{2},\ell,q,r] \subseteq
\PCP\left[
\varepsilon_c,\
\varepsilon_s'=\max\{7/8,\varepsilon_s+1/100\},
\Sigma'=\F_{2},\
\ell'=2^{\ell},\
q'=3+2tq,\
r'=r+(2+t)\ell
\right].
\]
Here \(t=8(\lfloor\log_2(100q)\rfloor+1)\) is the shift count from
\ref{LPCPToPCP.numShifts}.
\end{lemma}
\begin{proof}\uses{LPCPToPCP.verifier_completeness,LPCPToPCP.verifier_soundness_of_cases,LPCPToPCP.numShifts}\leanok
% Lecture 08 states the reduction for arbitrary finite fields. The formalized
% theorem here is the \(\F_2\) instance, so the proof length is \(2^\ell\) and
% the \(\log|\F|\) randomness factors disappear.
Use the verifier \(\overline V\) from \ref{LPCPToPCP.verifier}. Completeness is
\ref{LPCPToPCP.verifier_completeness}, and soundness is
\ref{LPCPToPCP.verifier_soundness_of_cases}. The BLR test makes three proof
queries. Each simulated LPCP query uses two PCP queries for each of the \(t\)
shifts, so the query bound is \(3+2tq\). The randomness consists of the
original \(r\) random bits, \(2\ell\) bits for the BLR test, and \(t\ell\) bits
for the shifts.
\end{proof}

\begin{definition}
\label{QESAT.linearization}
\uses{QESAT}
Let \(x=(p_1,\dots,p_m)\) be a \(\QESAT(\F_2,n)\) instance. For each quadratic
polynomial \(p_i\), form a row \(M_i\in\F_2^{n+n^2}\) by replacing every
linear monomial \(X_j\) by the variable \(a_j\) and every quadratic monomial
\(X_jX_k\) by the tensor variable \(b_{jk}\). Let
\[
  M_x :=
  \begin{bmatrix} M_1 \\ \vdots \\ M_m \end{bmatrix},
  \qquad
  c_x := (-\operatorname{const}(p_1),\dots,-\operatorname{const}(p_m)).
\]
Then \(M_x(a\parallel b)=c_x\) expresses the original equations after
replacing \(a_ja_k\) by \(b_{jk}\).
\end{definition}

\begin{definition}
\label{QESAT.verifier}
\lean{QESAT.verifier}
\uses{LPCPVerifier,QESAT.linearization,LINEQ.verifier,TENSORQ.verifier}
\leanok
The LPCP verifier for \(\QESAT(\F_2,n)\) has oracle access to a proof
\(\pi=a\parallel b\in\F_2^{n+n^2}\). On input \(x=(p_1,\dots,p_m)\), it first
rejects if some \(p_i\) has total degree \(>2\). Otherwise it constructs
\((M_x,c_x)\) as in \ref{QESAT.linearization} and accepts if both of the
following checks accept:
\begin{enumerate}
  \item the LINEQ verifier \ref{LINEQ.verifier} for \(M_x\pi=c_x\);
  \item the tensor self-check which queries \(\pi\) at
    \((s,0)\), \((t,0)\), and \((0,\mathrm{flat}(s\otimes t))\) and checks
    \[
      \langle b,\mathrm{flat}(s\otimes t)\rangle
      =\langle a,s\rangle\langle a,t\rangle .
    \]
\end{enumerate}
\end{definition}

\begin{lemma}
\label{QESAT.verifier_completeness}
\lean{QESAT.verifier_correct}
\uses{QESAT.verifier,QESAT.linearization,LINEQ.verifier,TENSORQ.tensor_check_complete}
\leanok
If \(x=(p_1,\dots,p_m)\in \QESAT(\F_2,n)\), then there is a proof
\(\pi\in\F_2^{n+n^2}\) accepted by \ref{QESAT.verifier} with probability \(1\).
\end{lemma}
\begin{proof}
Let \(a\in\F_2^n\) satisfy all equations and set
\(b=\mathrm{flat}(a\otimes a)\). By construction of
\ref{QESAT.linearization}, \(M_x(a\parallel b)=c_x\), so the LINEQ check
accepts for every \(r\). The tensor self-check accepts for every \(s,t\) by
\ref{TENSORQ.tensor_check_complete}.
\end{proof}

\begin{lemma}
\label{QESAT.verifier_soundness}
\lean{QESAT.verifier_correct}
\uses{QESAT.verifier,QESAT.linearization,LINEQ.linear_form_uniform_prob_mul_card_le_one,TENSORQ.verifier_soundness_after_sampling}
\leanok
If \(x=(p_1,\dots,p_m)\notin \QESAT(\F_2,n)\), then every proof
\(\pi\in\F_2^{n+n^2}\) is accepted by \ref{QESAT.verifier} with probability at
most \(3/4\).
\end{lemma}
\begin{proof}
If some \(p_i\) has total degree \(>2\), the verifier rejects immediately, so
the acceptance probability is \(0\). Assume now that every \(p_i\) is quadratic
and write \(\pi=a\parallel b\). If \(b\neq\mathrm{flat}(a\otimes a)\), then
the tensor self-check accepts with probability at most \(3/4\) by
\ref{TENSORQ.verifier_soundness_after_sampling} specialized to \(\F_2\), and
the full verifier accepts only if that check accepts.

Otherwise \(b=\mathrm{flat}(a\otimes a)\). Since \(x\notin\QESAT(\F_2,n)\),
the assignment \(a\) does not satisfy all equations. By
\ref{QESAT.linearization}, \(M_x(a\parallel b)\neq c_x\). The LINEQ check
therefore accepts with probability at most \(1/|\F_2|=1/2\) by
\ref{LINEQ.linear_form_uniform_prob_mul_card_le_one}. Again the full verifier
accepts only if this check accepts, so the acceptance probability is at most
\(1/2\leq 3/4\).
\end{proof}

\begin{theorem}
\label{QESAT_poly_LPCP}
\lean{QESAT_poly_LPCP}
\uses{QESAT, LPCP, QESAT.verifier}
\leanok
\[
  \mathrm{QESAT}(\F_2,n) \in
  \mathrm{LPCP}[
    \varepsilon_c = 0,
    \varepsilon_s = 3/4,
    \Sigma = \F_2,
    \ell = n + n^2,
    q = 4,
    r = |x| + 2n
  ].
\]
\end{theorem}
\begin{proof}\uses{QESAT.verifier_completeness,QESAT.verifier_soundness}\leanok
% Lecture 08 gives the exact LINEQ randomness as \(m\log|\F|\). In Lean,
% \(|x|\) is the encoded input size and bounds the number of equations \(m\);
% after specializing to \(\F_2\), this yields the displayed \(|x|+2n\) bound.
Use the verifier \ref{QESAT.verifier}. It makes one query for the LINEQ check
and three queries for the tensor self-check, for a total of four. Its
randomness is the LINEQ randomness plus two vectors in \(\F_2^n\). Completeness
and soundness are \ref{QESAT.verifier_completeness} and
\ref{QESAT.verifier_soundness}.
\end{proof}

\begin{theorem}
\label{QESAT_exp_PCP_before_repetition}
\lean{QESAT_exp_PCP_before_repetition}
\uses{QESAT,PCP}
\leanok
\[
  \mathrm{QESAT}(\F_2,n) \in
  \mathrm{PCP}[
    \varepsilon_c = 0,
    \varepsilon_s = 7/8,
    \Sigma = \F_2,
    \ell = 2^{|x|},
    q = O(1),
    r = |x|^{O(1)}
  ].
\]
\end{theorem}
\begin{proof}\uses{QESAT_poly_LPCP,LPCP_to_PCP_ZMod2}\leanok
% The lecture's final no-abstraction slide shares local-correction work in a
% direct QESAT verifier and therefore displays different constants. This
% formalization deliberately routes through the modular LPCP-to-PCP theorem,
% so only the asymptotic constant-query statement is retained here.
Apply \ref{LPCP_to_PCP_ZMod2} to \ref{QESAT_poly_LPCP}. Since the LPCP
soundness is \(3/4\), the converted soundness is
\[
  \max\{7/8,\,3/4+1/100\}=7/8.
\]
The query bound remains constant because \(q=4\), and the randomness is
polynomial in the input size. The Lean formalization pads the proof oracle from
length \(2^{n+n^2}\) to length \(2^{|x|}\).
\end{proof}

\begin{lemma}
\label{QESAT.constant_repetition}
\uses{PCP}
If a language is in
\(\PCP[0,7/8,\F_2,2^{|x|},O(1),|x|^{O(1)}]\), then it is also in
\(\PCP[0,1/2,\F_2,2^{|x|},O(1),|x|^{O(1)}]\).
\end{lemma}
\begin{proof}
% This repetition is folded into the Lean theorem `QESAT_exp_PCP` rather than
% exposed as a standalone declaration.
Run the \(7/8\)-sound verifier independently six times and accept only if all
six runs accept. Completeness error remains \(0\). For \(x\notin L\), the
acceptance probability is at most \((7/8)^6\leq 1/2\). Query and randomness
bounds are multiplied by the constant \(6\).
\end{proof}

\end{proof}
