\begin{lemma}[Chernoff bound]
\label{chernoff-bound}
\lean{LPCPToPCP.chernoff_bound}
\leanok
  Let $X_1,\dots, X_N$ be independent random variables over $\{0,1\}$, and let $\mu=\EE\left[\sum_{i = 1}^N X_i\right]$. For any $\Delta > 0$ one has
  \begin{equation*}
    \Pr\left[\sum_{i = 1}^N X_i \ge \mu+\Delta\right] \le \exp(-2\Delta^2/N)
  \end{equation*}
\end{lemma}
\begin{proof}
  Write $p_i=\EE[X_i]$, so that $\mu=\sum_{i=1}^N p_i$. We first prove the elementary exponential-moment estimate
  \[
    \EE[\exp(\lambda(X_i-p_i))]\le \exp(\lambda^2/8)
  \]
  for every $\lambda>0$ and every $i$. Since $X_i$ is $\{0,1\}$-valued,
  \[
    \EE[\exp(\lambda(X_i-p_i))]
    = (1-p_i)\exp(-\lambda p_i)+p_i\exp(\lambda(1-p_i)).
  \]
  Equivalently, it is enough to show
  \[
    \log(1-p_i+p_i\exp(\lambda))-p_i\lambda \le \lambda^2/8.
  \]
  For $p\in[0,1]$, set $f(\lambda)=\log(1-p+p\exp(\lambda))-p\lambda$. Then $f(0)=f'(0)=0$, and
  \[
    f''(\lambda)
    = \frac{p\exp(\lambda)}{1-p+p\exp(\lambda)}
      \left(1-\frac{p\exp(\lambda)}{1-p+p\exp(\lambda)}\right)
    \le \frac14.
  \]
  Hence, for $\lambda>0$,
  \[
    f(\lambda)
    = \int_0^\lambda\int_0^u f''(v)\,dv\,du
    \le \int_0^\lambda\int_0^u \frac14\,dv\,du
    = \lambda^2/8.
  \]

  Let $S=\sum_{i=1}^N X_i$. By independence and the estimate above,
  \[
    \EE[\exp(\lambda(S-\mu))]
    = \prod_{i=1}^N \EE[\exp(\lambda(X_i-p_i))]
    \le \exp(N\lambda^2/8).
  \]
  Markov's inequality gives, for every $\lambda>0$,
  \[
    \Pr[S\ge \mu+\Delta]
    = \Pr[\exp(\lambda(S-\mu))\ge \exp(\lambda\Delta)]
    \le \exp(-\lambda\Delta)\EE[\exp(\lambda(S-\mu))]
  \]
  Using the moment bound above, this is at most
  \[
    \exp(-\lambda\Delta+N\lambda^2/8).
  \]
  Taking $\lambda=4\Delta/N$ gives
  \[
    \Pr[S\ge \mu+\Delta]\le \exp(-2\Delta^2/N).
  \]
\end{proof}

\begin{definition}
\label{num-shifts}
\uses{LPCPVerifier}
\lean{LPCPToPCP.numShifts}
\leanok
For an LPCP verifier making at most \(q\) linear queries, set
\[
  t := 8\bigl(\lfloor \log_2(100q)\rfloor+1\bigr).
\]
% Lecture 08 only needs \(t=O(\log q)\). The Lean proof fixes this base-two
% value because the formalization uses `Nat.log2`; it is stronger than needed
% for \(q\exp(-t/8)\le 1/100\).
This is the concrete logarithmic shift count used in Lean. The converted PCP
verifier uses \(t\) independent random shifts to decode each linear query by
plurality.
\end{definition}

\begin{definition}[Encoded proof]
\label{LPCPToPCP.encodedProof}
% TODO: define LPCP proofs and add it to \uses
\lean{LPCPToPCP.encodedProof}
\uses{PCPVerifier}
\leanok
For a linear LPCP proof \(\pi\in\F_2^\ell\), its PCP encoding \(\widehat{\pi}_{\mathrm{lin}} \in \F_2^{\F_2^\ell}\) is the truth
table
\[
  \widehat{\pi}_{\mathrm{lin}}(a) := \langle \pi,a\rangle,
  \qquad a\in\F_2^\ell.
\]
\end{definition}

\begin{definition}
\label{LPCPToPCP.correctedAnswer}
% TODO: define PCP proofs and add it to \uses
\lean{LPCPToPCP.correctedAnswer,LPCPToPCP.decodedLinearQuery,LPCPToPCP.pluralityZMod2}
\uses{num-shifts}
\leanok
Given a PCP proof \(\widehat{\pi}:\F_2^\ell\to\F_2\), shifts
\(r_1,\dots,r_t\in\F_2^\ell\), and a linear query \(a\in\F_2^\ell\), define
the locally corrected answer
\[
  \mathsf{Corr}_{\widehat{\pi}}(a;r_1,\dots,r_t)
  :=
  \operatorname{plurality}_{i\in[t]}
  \bigl(\widehat{\pi}(a+r_i)-\widehat{\pi}(r_i)\bigr).
\]
\end{definition}

\begin{definition}
\label{LPCPToPCP.verifier}
\lean{LPCPToPCP.verifier}
\uses{PCPVerifier,LPCPVerifier,BLR.basicVerifier,LPCPToPCP.correctedAnswer}
\leanok
Let \(V\) be an LPCP verifier over \(\F_2\). The converted PCP verifier
\(\overline V\), with oracle access to
\(\widehat{\pi}:\F_2^\ell\to\F_2\), proceeds as follows on input \(x\):
\begin{enumerate}
  \item If \(V_{\mathrm{BLR}(\F_{2}, n)}^{\widehat{\pi}}=0\), reject. Otherwise, proceed to the next step.
  \item Sample \(t\) shifts \(r_1,\dots,r_t\in\F_2^\ell\), with \(t\) as in
    \ref{LPCPToPCP.numShifts}.
  \item Simulate \(V(x)\). Whenever \(V\) asks a linear query
    \(a\in\F_2^\ell\), answer it by
    \(\mathsf{Corr}_{\widehat{\pi}}(a;r_1,\dots,r_t)\).
\end{enumerate}

\begin{lemma}[Completeness]
\label{LPCPToPCP.verifier_completeness}
\lean{LPCPToPCP.verifier_completeness}
\uses{LPCPToPCP.verifier,LPCPToPCP.encodedProof,BLR_basic_completeness_ZMod2,LPCP}
\leanok
If \(\Pr\!\left[V^{\langle \pi, \cdot \rangle}(x)=1\right]\ge 1-\varepsilon_c\), then \(\Pr\!\left[\overline{V}^{\widehat{\pi}_{\mathrm{lin}}}(x)=1\right]\ge 1-\varepsilon_c\).
\end{lemma}
\begin{proof}\leanok
The encoded proof is linear, so the BLR test accepts with probability \(1\) by
\ref{BLR_basic_completeness_ZMod2}. For every query \(a\) and every shift
\(r_i\),
\[
  \widehat{\pi}_{\mathrm{lin}}(a+r_i)-\widehat{\pi}_{\mathrm{lin}}(r_i)
  = \langle \pi,a\rangle.
\]
Thus every plurality answer equals the corresponding LPCP oracle answer, and
the simulation has exactly the same acceptance probability as \(V\).
\end{proof}


\begin{lemma}[Soundness, far case]
\label{LPCPToPCP.verifier_soundness_far_case_of_isFar}
\lean{LPCPToPCP.verifier_soundness_far_case_of_isFar}
\uses{LPCPToPCP.verifier,BLR_basic_soundness_ZMod2}
\leanok
If \(\distanceToLin{\widehat{\pi}} \geq  1 / 8\), then
\[
  \Pr[\overline V^{\widehat{\pi}}(x)=1]\leq 7/8 .
\]
\end{lemma}
\begin{proof}\leanok
By \ref{BLR_basic_soundness_ZMod2}, we know that \(\Pr [V_{\text{BLR}}^{\bar{\pi }}=0]\ge 1/8\). Since \(\overline V\) accepts only when this test
accepts, its acceptance probability is at most \(7/8\).
\end{proof}

\begin{lemma}
\label{LPCPToPCP.local_correction}
\lean{LPCPToPCP.localCorrectionBad_single_le_quarter_of_not_far,LPCPToPCP.correctedAnswer_failure_le_exp_of_single,LPCPToPCP.chernoff_half_of_mean_le_quarter,LPCPToPCP.numShifts_eta_bound}
\uses{LPCPToPCP.correctedAnswer,chernoff-bound}
\leanok
If \(\distance{\widehat{\pi}}{f} < 1 / 8\) for some linear
function \(f\) of the form \(u\mapsto\langle\pi,u\rangle\), then
\[
  \Pr_{r_1,\dots,r_t}\!\left[
    \mathsf{Corr}_{\widehat{\pi}}(a;r_1,\dots,r_t)\neq \langle\pi,a\rangle
  \right]
  \leq \exp(-t/8), \quad \forall a\in\F_2^\ell.
\]
\end{lemma}
\begin{proof}\leanok
For one shift \(r\), the corrected sample can be wrong only if
\(\widehat{\pi}(a+r)\) or \(\widehat{\pi}(r)\) disagrees \(f\). A union bound gives error probability at most
\(2\cdot(1/8)=1/4\). If \(X_i\) is the indicator of failure for shift \(r_i\),
then \(\EE[\sum_i X_i]\le t/4\). The plurality answer is wrong only if
\(\sum_i X_i\ge t/2\), so \ref{LPCPToPCP.chernoff_bound} with
\(\Delta=t/4\) gives \(\exp(-t/8)\).
% Lecture 07 states a looser concentration estimate with different constants.
% The blueprint follows the Hoeffding normalization used by the Lean lemma
% `chernoff_half_of_mean_le_quarter`, which is mathematically stronger here.
\end{proof}

\begin{lemma}
\label{LPCPToPCP.corrected_simulation_hybrid}
\lean{LPCPToPCP.corrected_run_accept_le_linear_add,LPCPToPCP.verifier_accept_le_linear_add_of_query_correction}
\uses{LPCPToPCP.verifier,LPCPToPCP.local_correction,LPCP}
\leanok
Assume every possible linear query \(a\) is locally corrected incorrectly with
probability at most \(\eta\). If the simulated LPCP computation makes at most
\(q\) proof queries, then its acceptance probability under the corrected PCP
oracle is at most the acceptance probability under the nearby linear proof
plus \(q\eta\).
\end{lemma}
\begin{proof}\leanok
Expose the LPCP computation one oracle query at a time. At a proof query, the
two simulations can diverge only when the local correction for the queried
linear form is wrong; by assumption this has probability at most \(\eta\).
Inducting over the query-bound derivation gives an additive loss of at most
\(q\eta\).
% Lecture 08 presents this step as a union bound over the verifier's \(q\)
% queries. The Lean proof phrases the same estimate as a sequential hybrid,
% which also covers adaptive LPCP verifiers whose later queries may depend on
% earlier corrected answers.
\end{proof}

\begin{lemma}
\uses{LPCP,PCP,num-shifts}
\label{LPCP_to_PCP_ZMod2}
\lean{LPCP_to_PCP_ZMod2}
\leanok
\begin{gather*}
\LPCP[\varepsilon_c,\varepsilon_s,\Sigma=\F_{2},\ell,q,r] \subseteq\\
\PCP\left[
\varepsilon_c,\
\varepsilon_s'=\max\{7/8,\varepsilon_s+1/100\},
\Sigma'=\F_{2},\
\ell'=2^{\ell},\
q'=3+2tq,\
r'=r+(2+t)\ell
\right].
\end{gather*}
\end{lemma}
\begin{proof}
\uses{LPCPToPCP.verifier, LPCPToPCP.verifier_completeness, LPCPToPCP.verifier_soundness_of_cases}
Let $L \in \LPCP[\varepsilon_c,\varepsilon_s,\Sigma=\F_{2},\ell,q,r]$ and let $V$ be the LPCP verifier for $L$. Use the converted PCP verifier $\overline V$ defined in \ref{LPCPToPCP.verifier}.
Completeness is
\ref{LPCPToPCP.verifier_completeness}, and soundness is
\ref{LPCPToPCP.verifier_soundness_of_cases}. The BLR test makes three proof
queries. Each simulated LPCP query uses two PCP queries for each of the \(t\)
shifts, so the query bound is \(3+2tq\). The randomness consists of the
original \(r\) random bits, \(2\ell\) bits for the BLR test, and \(t\ell\) bits
for the shifts.
\end{proof}
